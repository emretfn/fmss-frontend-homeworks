```bash
npm install --save fmss-howework
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'fmss-howework'
import 'fmss-howework/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
